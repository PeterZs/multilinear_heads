24 static bool load_matrix(Eigen::MatrixXd& eigenMatrix,
25  const std::string& filename)
26 {
27  std::ifstream ifs( filename.c_str(), std::ofstream::binary );
28  if (!ifs.is_open())
29  {
30  std::cerr << "Cannot load matrix " << filename << std::endl;
31  return false;
32  }
33  unsigned int n_rows = 0;
34  unsigned int n_cols = 0;
35  ifs.read(reinterpret_cast<char *>(&n_rows), sizeof(n_rows));
36  ifs.read(reinterpret_cast<char *>(&n_cols), sizeof(n_cols));
37  eigenMatrix.resize(n_rows, n_cols);
38  ifs.read(reinterpret_cast<char *>(eigenMatrix.data()), n_rows*n_cols*sizeof(typename Eigen::MatrixXd::Scalar) );
39  ifs.close();
40 
41  return true;
42 }

68 static bool load_scalars(std::vector<double>& output, const std::string& filename)
69 {
70  std::ifstream ifs(filename);
71  if(!ifs.is_open())
72  {
73  std::cerr << "Cannot read weights from " << filename << std::endl;
74  return false;
75  }
76 
77  output.clear();
78 
79  double val;
80  while (true)
81  {
82  ifs >> val;
83  if (ifs.eof()) break;
84  output.push_back(val);
85  }
86 
87  ifs.close();
88 
89  return (!output.empty());

95 static bool load_parameters(Eigen::VectorXd& wSkull,
96  Eigen::VectorXd& wFstt,
97  const std::string& filenamewSkull,
98  const std::string& filenamewFstt)
99 {
100  std::vector<double> vecwSkull;
101  if (!load_scalars(vecwSkull, filenamewSkull))
102  {
103  std::cerr << "[ERROR] in 'load_parameters(...)' - Can't load w_skull" << std::endl;
104  return false;
105  }
106 
107  std::vector<double> vecwFstt;
108  if (!load_scalars(vecwFstt, filenamewFstt))
109  {
110  std::cerr << "[ERROR] in 'load_parameters(...)' - Can't load w_fstt" << std::endl;
111  return false;
112  }
113 
114  if ((int)wSkull.size() > (int)vecwSkull.size() ||
115  (int)wFstt.size() > (int)vecwFstt.size())
116  {
117  std::cerr << "[ERROR] in 'load_parameters(...)' - Loaded w_skull and w_fstt are too short" << std::endl;
118  return false;
119  }
120  if (wSkull.size() == 0 || wFstt.size() == 0)
121  {
122  std::cerr << "[ERROR] in 'load_parameters(...)' - No previously loaded parameters w_skull and w_fstt available" << std::endl;
123  return false;
124  }
125 
126  for (unsigned int i = 0; i < wSkull.size(); ++i)
127  {
128  wSkull(i) = vecwSkull[i];
129  }
130 
131  for (unsigned int i = 0; i < wFstt.size(); ++i)
132  {
133  wFstt(i) = vecwFstt[i];
134  }
135 
136  return true;
137 }
